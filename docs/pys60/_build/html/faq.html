>>> def ru(text):
        return text.decode('кодировка файла скрипта')
>>> text = ru('привет')

>>> # в начале скрипта создаем замок
>>> lock = e32.Ao_lock()
>>> ...
>>> def exit():

>>> ...
>>> # в конце скрипта
>>> lock.wat()
