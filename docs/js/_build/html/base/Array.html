Array - массивы¶

class Array()¶
class Array(размер)
class Array(элемент0, элемент1, ...)

Массив

Наследник Object()

var array = [];

var cities = ['moscow', 'kazan'];
cities[0];
// 'moscow'

cities[3] = 'abakan';
cities.length;
// 4

cities;
//['moscow', 'kazan', '', 'abakan']

cities.length = 2;
cities;
// ['moscow', 'kazan']

cities['city'] = 45;
cities.length;
// 2

var a = new Array();

var b = new Array('hey', 'you');

var c = new Array(3);

var a = [];
a[5] = 5;
for (var x in a){
    // выведет только 5
}
for (var i=0; i<a.length; i++){
    // выведет все 5 элементов
}
length¶

Возвращает число, количество элементов в массиве

['moscow', 'kazan'].length;
// 2
concat(значение, ...)¶

Возвращает новый массив, расширенный значениями из аргумента

var a = [1, 2, 3];

a.concat([4, 5], 'end');
// [1, 2, 3, 4, 5, 'end']

a.concat([4, 5])
// [1, 2, 3, 4, 5]
every(callback[, this])¶

Возвращает булево, соответсвие всех элементов массива условию обработчика.

New in version ECMAScript5.

[1, 2, 3].every(function(x){ return x < 5});
// true

[1, 2, 3].every(function(x){ return x < 3});
// false
filter(callback[, filter])¶

Возвращает массив элементов, удовлетворяющих требованиям обработчика

New in version ECMAScript5.

[1,2,3].filter(function(x) { return x > 1; });
// [2, 3]
forEach(callback[, this])¶

Вызывает функцию-обработчик для каждого элемента массива

New in version ECMAScript5.

[1, 2, 3].forEach(function(array[i], i, array){
    ...
});
from(args)¶

New in version ECMAScript6.

var arr = Array.from(arguments);
join([str splitter=', '])¶

Возвращает строку, полученную преобразованием всех элементов массива в строки и сконкатенированные

New in version ECMAScript5.

[1,2,3].join('');
// '123'
indexOf(значение[, int pos=0])¶

Возвращает число, позиция элемента в массиве

New in version ECMAScript5.

['a','b','c'].indexOf('b');
// 1

['a','b','c'].indexOf('d');
// -1

['a','b','c'].indexOf('a', 1);
// -1
lastIndexOf(значение[, int pos=array.length])¶

Возвращает число, позиция элемента в массиве в обратном порядке

New in version ECMAScript5.

map(callback[, this])¶

Возвращает массив, вычисленный по функции-обработчику

New in version ECMAScript5.

[1,2,3].map(function(x) {
    return x*x;
});
// [1,4,9]
pop()¶

Возвращает последний элемент и удаляет его из массива

New in version ECMAScript5.

var c = [1,2,3];
c.pop();
// 3
c;
// [1,2]
push(значение1, ...)¶

Добавление объекта в массив в конец

var c = [1,2,3];
c.push(4);
c;
// [1,2,3,4]
reduce(callback[, int start=0])¶

Вычисляет значение на основе элементов данного массива, свертка массива

New in version ECMAScript5.

[1,2,3].reduce(function(a, b){
    return a + b;
});
// 6
reduceRight(callback[, int start=0])¶

Вычисляет значение на основе элементов данного массива, спарва налево, свертка массива

New in version ECMAScript5.

reverse(func, start)¶

Разворачивание массива

var a = [1, 2, 3];
a.reverse();
// [3, 2, 1]
shift()¶

Возвращает первый элемент массива, и удалеят его из массива

var a = [1, [2, 3], 4];
var b = a.shift();
// 1
a;
// [[2, 3], 4]
slice(start, end)¶

Возвращает фрагмент массива

[1, 2, 3, 4, 5].slice(0, 3);
// [1, 2, 3]

[1, 2, 3, 4, 5].slice(3);
// [4, 5]

[1, 2, 3, 4, 5].slice(1, -1);
// [2, 3, 4]

[1, 2, 3, 4, 5].slice(-3, -2);
// [3]
some(callback[, this])¶

Проверяет, возвращает ли предикат значение true хотя бы для одного элемента массива

New in version ECMAScript5.

[1,2,3].some(function(x) { return x > 5; }) // => false: нет эле­мен­тов > 5
[1,2,3].some(function(x) { return x > 2; }) // => true: не­ко­то­рые > 3
[].some(function(x) { return true; });      // => false: все­гда false для []
sort([sorter_func])¶

Сортирует массив

var a = [1, 2, 15];
a.sort();
a;
// [1, 15, 2]
splice(start, count, value1, value2, ...)¶

Возвращает массив. Вставляет, удаляет или заменят элементы массива

var c = [1, 2, 3, 4, 5];

c.splice(1,2);
// [2, 3]

c;
// [1, 4, 5];

c.splice(1, 2, 33, 44);
// [4, 5]

c;
// [1, 33, 44]
unshift(var1, var2 ...)¶

Добавляет в начало массива элементы

var a = [];
a.unshift(1);
a;
// [1]
