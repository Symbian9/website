Array - массивы¶

class Array()¶

Массив

var array = [];

var cities = ['moscow', 'kazan'];
cities[0];
// 'moscow'

cities[3] = 'abakan';
cities.length;
// 4

cities;
//['moscow', 'kazan', '', 'abakan']

cities.length = 2;
cities;
// ['moscow', 'kazan']

cities['city'] = 45;
cities.length;
// 2

var a = new Array();

var b = new Array('hey', 'you');

var c = new Array(3);

    // выведет только 5

concat(func, start)¶

Возвращает объединенный массив

Arguments:
  • func – функция, принимает два аргумента
  • start – начальное значение
 
var c = [1,2,3];
var d = [1,2,3];

var e = c.concat(d, 'end');

e;
// [1,2,3,1,2,3, 'end']
join([con])¶

Массив в строку

Arguments:
  • con – разделитель
 
var c = [1,2,3];

c.join('');
c;
// '123'
indexOf(val[, start])¶

Поиск элемента в массиве с начала

lastIndexOf(val[, start])¶

Поиск элемента в массиве с конца

pop(obj)¶

Возвращает последний элемент и удаляет его из массива

var c = [1,2,3];
c.pop();
// 3
c;
// [1,2]
push(obj)¶

Добавление объекта в массив в конец

var c = [1,2,3];
c.push(4);
c;
// [1,2,3,4]
reduce(func, start)¶

Обработка массива

Arguments:
  • func – функция принимает два аргумента
  • start – начальное значение
 
var c = [1,2,3];

function adder(a, b) { return a+b; }

c.reduce(adder, 0);
// 6

sort(func)¶

Сортировка массива

var c = [1, 2, 15];
c.sort();
// [1,15,2]

function compare(a,b){
    return a-b;
    // вернет положительное => поменять местами
    // отрицательно => все норм
}
c.sort();
// [1, 2, 15]
splice(start, count)¶

Удаление элементов в массиве

Arguments:
  • start – начальная позиция
  • count – количесвто удаляемых элементов
 
var с = [1, 2, 3, 4, 5];
c.splice(2,2);
// [3, 4]

c;
[1, 2, 5];
