Number - число¶

var a = 1,
    a = 1.23,
    a = 1E6,

    // восьмиричная форма записи
    a = 045,

    // шестнадцатеричная форма
    a = 0x45,

    a = 10 % 3,
    // 2, остаток от целочисленного деления

    a++;
    a--;
    ++a;
    --a;
    a += 5;
    a -= 5;
    a /= 5;

0.1 + 0.2 == 0.3;
// false

1 == 1.0;
// true

1/0;
// бесконечность, Infinity

-1/0 = -Infinity
// бесконечность

isInteger(number)¶

New in version ECMAScript6.

Number.isInteger(42);     // true
Number.isInteger(42.000); // true
Number.isInteger(42.3);   // false

if (!Number.isInteger){
    Number.isInteger = function(num){
        return (
            typeof num === 'number' &&
            num % 1 == 0
        );
    }
}
isNan(number)¶

New in version ECMAScript6.

if (!Number.isNan){
    Number.isNan = function(num){
        return (
            typeof num === 'number' &&
            window.isNan(num)
        );
    }
}
isSafeInteger(number)¶

New in version ECMAScript6.

Number.isSafeInteger(Number.MAX_SAFE_INTEGER); // true
Number.isSafeInteger(Math.pow(2, 53));         // false
Number.isSafeInteger(Math.pow(2, 53) - 1);     // true

if (!Number.isSafeInteger){
    Number.isSafeInteger = function(num){
        return (
            Number.isInteger(num) &&
            Math.abs(num) <= Number.MAX_SAFE_INTEGER
        );
    }
}
toExponential([offset])¶

Фор­ма­ти­ру­ет чис­ло в экс­по­нен­ци­аль­ную фор­му пред­став­ле­ния

var x = 123456789;

x.toExponential();
//'1.23456789e+8'

x.toExponential(1);
//'1.2e+8'

x.toExponential(2);
//'1.23e+8'

x.toExponential(3);
//'1.235e+8'
toFixed([offset])¶

Возвращает строку, с определенным количеством знаков после запятой

var y = 43.81327;

y.toFixed();
//'44'

y.toFixed(1);
//'43.8'

y.toFixed(2);
//'43.81'

y.toFixed(3);
//'43.813'
